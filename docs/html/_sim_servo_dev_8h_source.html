165 bool SimServo::_validateMap(void) {
166 
167  if (_mapPair < 2) {
168  return false;
169  }
170 
171  // each input must be greater than the previous input
172  for(unsigned int i = 1; i < _mapPair; ++i) {
173  if (_map[i][In] < _map[i-1][In]) {
174  return false;
175  }
176  }
177 
178  return true;
179 }

184 void SimServo::_update(bool updateOutput) {
185 
186  // assign dataref to stored 'input' value
187  _in = _dr;
188 
189  // if input off map, put output on edge of map
190  if (_in <= _map[0][In]) {
191  _out = _map[0][Out];
192  }
193 
194  if (_in >= _map[_mapPair-1][In]) {
195  _out = _map[_mapPair-1][Out];
196  }
197 
198  // if input within map, interpolate output
199  for (int i = 1; i < _mapPair; ++i) {
200  if (_in < _map[i][In]) {
201 
202  double buf = _in;
203  buf -= _map[i-1][In];
204  buf /= _map[i][In] - _map[i-1][In];
205  buf *= _map[i][Out] - _map[i-1][Out];
206  buf += _map[i-1][Out];
207  _out = buf;
208 
209  // exit for loop
210  i = _mapPair;
211  }
212  }
213 
214  // if we have power, or don't need power
215  if(*_powerSource || !_needsPower) {
216  // convert double to int to give to RC servo
217  _servoAngle = (int)(_out + 0.5);
218  } else {
219  // move to resting position if defined
220  if (_restAngle > -1) {
221  _servoAngle = _restAngle;
222  }
223  // otherwise servoAngle does not change
224  }
225 
226  // use updateOutput as a final gate to write to servo
227  if (updateOutput && FlightSim.isEnabled()) {
228  _servo.write(_servoAngle);
229  }
230 
231  return;
232 }
