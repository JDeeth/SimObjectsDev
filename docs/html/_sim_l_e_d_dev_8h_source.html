181 void SimLED::addToLinkedList() {
182  next_ = 0;
183 
184  if (count_++ == 0) {
185  first_ = this;
186  } else {
187  // Go through linked list and make last existing element point to us
188  SimLED *a = first_;
189  while (a->next_)
190  a = a->next_;
191  a->next_ = this;
192  }
193 }

205 void SimLED::update( bool updateOutput) {
206 
207  if (count_ > 0) {
208  SimLED* buf = first_;
209  while (buf != 0) {
210  buf->update_(updateOutput);
211  buf = buf->next_;
212  }
213  }
214 }

216 // Determine whether this SimLED should be lit
217 void SimLED::update_(bool updateOutput) {
218 
219  switch(type) {
220  case SLInt:
221  active_ = (lowLimitI_ <= _drI && _drI <= highLimitI_);
222  break;
223  case SLFloat:
224  active_ = (lowLimitF_ <= _drF && _drF <= highLimitF_);
225  break;
226  }
227 
228  if(inverse_ == true)
229  active_ = !active_;
230 
231  lit_ = active_;
232 
233  // apply lit_ filters:
234 
235  // we are lit if bulb-test is active
236  if( (allowTest_ == true) && (testAll_ == true) )
237  lit_ = true;
238 
239  // we are not lit if the sim isn't running or no simulated power
240  if( (FlightSim.isEnabled() == false) || (hasPower_ == false) ) {
241  lit_ = false;
242  }
243 
244  // unless ordered otherwise, light or extinguish LED based on our lighting state
245  if (updateOutput)
246  digitalWrite(_pin, lit_);
247 }
