201 bool SimServo::_validateMap(void) {
202 
203  if (_mapPair < 2) {
204  return false;
205  }
206 
207  // each input must be greater than the previous input
208  for(unsigned int i = 1; i < _mapPair; ++i) {
209  if (_map[i][In] < _map[i-1][In]) {
210  return false;
211  }
212  }
213 
214  return true;
215 }

220 void SimServo::_update(bool updateOutput) {
221 
222  // assign dataref to stored 'input' value
223  _in = _dr;
224 
225  // if input off map, put output on edge of map
226  if (_in <= _map[0][In]) {
227  _out = _map[0][Out];
228  }
229 
230  if (_in >= _map[_mapPair-1][In]) {
231  _out = _map[_mapPair-1][Out];
232  }
233 
234  // if input within map, interpolate output
235  for (int i = 1; i < _mapPair; ++i) {
236  if (_in < _map[i][In]) {
237 
238  double buf = _in;
239  buf -= _map[i-1][In];
240  buf /= _map[i][In] - _map[i-1][In];
241  buf *= _map[i][Out] - _map[i-1][Out];
242  buf += _map[i-1][Out];
243  _out = buf;
244 
245  // exit for loop
246  i = _mapPair;
247  }
248  }
249 
250  // if we have power, or don't need power
251  if(*_powerSource || !_needsPower) {
252  // convert double to int to give to RC servo
253  _servoAngle = (int)(_out + 0.5);
254  } else {
255  // move to resting position if defined
256  if (_restAngle > -1) {
257  _servoAngle = _restAngle;
258  }
259  // otherwise servoAngle does not change
260  }
261 
262  // use updateOutput as a final gate to write to servo
263  if (updateOutput && FlightSim.isEnabled()) {
264  _servo.write(_servoAngle);
265  }
266 
267  return;
268 }

273 void SimServo::_addToLinkedList(void) {
274 
275  _next = 0;
276 
277  if (_first == 0) {
278  _first = this;
279  } else {
280  // Go through linked list and make last existing element point to us
281  SimServo *a = _first;
282  while (a->_next)
283  a = a->_next;
284  a->_next = this;
285  }
